Introduction To Python | Classes

#Classes are defined using the class keyboard

class Calc(object):
    # Initializes num1 and num2
    def __init__(self, num1, num2):
        self.num1 = num1
        self.num2 = num2

    def sum(self):
        # Prints sum of initialized values
        print(self.num1+self.num2)

    def subs(self):
        # Prints substraction of initialized values
        print(self.num1-self.num2)

calc = Calc(10,5)
calc.sum() # = 15
calc.subs() # = 5
